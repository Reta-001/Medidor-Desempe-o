\section*{Subsistema 1 — Entrada ($S_C$) — M/M/2}

% Ecuación 1.1 — Factor de utilización del sistema
\[
\rho_{S_C} = \frac{\lambda_{S_C}}{c \cdot \mu_{S_C}} \tag{1.1}
\]

% Ecuación 1.2 — Probabilidad de que el sistema esté vacío
\[
P_0^{S_C} = \left[ \sum_{n=0}^{c-1} \frac{(c\rho_{S_C})^n}{n!} + \sum_{n=c}^{K} \frac{(c\rho_{S_C})^c}{c!} \cdot \rho_{S_C}^{n-c} \right]^{-1} \tag{1.2}
\]

% Ecuación 1.3 — Probabilidad de tener n clientes (n < c)
\[
P_n^{S_C} = \frac{(c\rho_{S_C})^n}{n!} \cdot P_0^{S_C} \quad \text{para } 0 \leq n < c \tag{1.3}
\]

% Ecuación 1.4 — Probabilidad de tener n clientes (n ≥ c)
\[
P_n^{S_C} = \frac{(c\rho_{S_C})^c}{c!} \cdot \rho_{S_C}^{n-c} \cdot P_0^{S_C} \quad \text{para } c \leq n \leq K \tag{1.4}
\]

% Ecuación 1.5 — Probabilidad de bloqueo o sistema lleno
\[
P_K^{S_C} = \frac{(c\rho_{S_C})^K}{c! \cdot c^{K-c}} \cdot P_0^{S_C} \tag{1.5}
\]

% Ecuación 1.6 — Tasa efectiva de llegada
\[
\lambda_{\text{ef}}^{S_C} = \lambda_{S_C} \cdot (1 - P_K^{S_C}) \tag{1.6}
\]

% Ecuación 1.7 — Número promedio de clientes en el sistema
\[
L_s^{S_C} = \sum_{n=1}^{K} n \cdot P_n^{S_C} \tag{1.7}
\]

% Ecuación 1.8 — Número promedio de servidores ocupados
\[
c_{\text{ocupados}}^{S_C} = \frac{\lambda_{\text{ef}}^{S_C}}{\mu_{S_C}} \tag{1.8}
\]

% Ecuación 1.9 — Número promedio de clientes en la cola (balance por componentes)
\[
L_q^{S_C} = L_s^{S_C} - \frac{\lambda_{\text{ef}}^{S_C}}{\mu_{S_C}} \tag{1.9}
\]

% Ecuación 1.10 — Relación directa de clientes en cola
\[
L_q^{S_C} = \sum_{n=c}^{K} (n - c) \cdot P_n^{S_C} \tag{1.10}
\]

% Ecuación 1.11 — Tiempo promedio de espera en la cola
\[
W_q^{S_C} = \frac{L_q^{S_C}}{\lambda_{\text{ef}}^{S_C}} \tag{1.11}
\]

% Ecuación 1.12 — Tiempo promedio total en el sistema
\[
W_s^{S_C} = W_q^{S_C} + \frac{1}{\mu_{S_C}} \tag{1.12}
\]
